<?php
/*
|--------------------------------------------------------------------------
| 网站路由设置
|--------------------------------------------------------------------------
|
| 路由必须按下面方式定义在路由组中
| 路由组必须存在 prefix 与 as 并且必须为 模块小写标识
*/
Route::group(['prefix' => '{lower-module}', 'as' => '{lower-module}.'], function () {
    Route::resource('test', '{module}Controller');
});

/*
|--------------------------------------------------------------------------
| 后台管理: 后台管理员操作的动作
|--------------------------------------------------------------------------
*/
Route::group([
    'prefix' => '{lower-module}/admin',
    'as' => '{lower-module}.admin.',
    'namespace' => 'Admin',
    'middleware' => ['admin'],
], function () {
    Route::resource('cms', 'CmsController');
});
/*
|--------------------------------------------------------------------------
| 会员中心: 会员中心使用的动作
